!-- OnEvent RoundStarted

# Шансы для спавна 3114
if {Chance 15%} is false
    stop
end

# Ждем 2 секунды, чтобы сервер успел выдать всем роли и заспавнить игроков
wait 1s

# Проверяем, есть ли вообще SCP в этом раунде
if {AmountOf @scpPlayers} is 0
    stop
end

# Начинаем поиск случайного SCP, который не является 079
@target = @empty

# Повторяем случайную выборку до 20 раз, чтобы гарантированно найти не 079
repeat 20
    @potTarget = LimitPlayers @scpPlayers 1
    
    # Если выбранный игрок не 079, сохраняем его и прерываем цикл
    if {@potTarget -> role} isnt "Scp079"
        @target = @potTarget
        break
    end
end

# Если после 20 попыток мы так и не нашли подходящего игрока
# (например, за SCP играет только один человек и ему выпал 079), останавливаем скрипт
if {AmountOf @target} is 0
    stop
end

# Выдаем выбранному игроку роль SCP-3114
SetRole @target Scp3114

# Ждем 1 секунду перед объявлением
wait 1s

# Запускаем сообщение CASSIE с джинглом (звуком перед началом)
# CASSIE прочитает "scp 3 1 1 4", а игроки на экране увидят перевод
Cassie jingle "scp 3 1 1 4 containment breach detected" "Обнаружено нарушение особых условий содержания SCP-3114"
