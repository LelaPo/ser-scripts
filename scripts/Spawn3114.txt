!-- OnEvent RoundStarted

# 1. Задержка для корректного спавна
wait 2s

# 2. Шанс
if {Chance 15%}
    
    # 3. Берем всех текущих SCP
    @allScps = @scpPlayers
    
    # 4. Исключаем из них всех, кто является SCP-079
    # Используем готовую системную переменную @scp079Players
    @candidates = RemovePlayers @allScps @scp079Players
    
    # Если на сервере были ТОЛЬКО компьютеры, список станет пустым.
    # В таком случае берем всех SCP (включая 079), чтобы раунд не остался без скелета.
    if {AmountOf @candidates} is 0
        @candidates = @allScps
    end

    # 5. Выбираем одну случайную цель
    @victim = LimitPlayers @candidates 1
    
    # 6. Если цель найдена — превращаем
    if {AmountOf @victim} > 0
        
        SetRole @victim Scp3114
        
        # Анонс
        wait 2s
        Cassie jingle "pitch_0.15 .g4 pitch_0.9 SCP 3 1 1 4 HAS SUCCESSFULLY BREACHED CONTAINMENT" "SCP-3114 успешно нарушил условия содержания."
        
    end
end
